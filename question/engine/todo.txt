-- Monster query to return all of a question_usage.

SELECT
    qasd.id,
    quba.id AS qubaid,
    quba.contextid,
    quba.component,
    quba.preferredbehaviour,
    qa.id AS questionattemptid,
    qa.questionusageid,
    qa.slot,
    qa.behaviour,
    qa.questionid,
    qa.maxmark,
    qa.minfraction,
    qa.flagged,
    qa.questionsummary,
    qa.rightanswer,
    qa.responsesummary,
    qa.timemodified,
    qas.id AS attemptstepid,
    qas.sequencenumber,
    qas.state,
    qas.fraction,
    qas.timecreated,
    qas.userid,
    qasd.name,
    qasd.value

FROM git_question_usages quba
LEFT JOIN git_question_attempts qa ON qa.questionusageid = quba.id
LEFT JOIN git_question_attempt_steps qas ON qas.questionattemptid = qa.id
LEFT JOIN git_question_attempt_step_data qasd ON qasd.attemptstepid = qas.id

WHERE
    quba.id = 1

ORDER BY
    qa.slot,
    qas.sequencenumber
